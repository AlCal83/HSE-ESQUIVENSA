Esta encuesta hace parte de la estructura del Sistema de Gestión en Seguridad y Salud en el Trabajo (SG-SST) de Especialidades Químicas Industriales SAS (EsquivenSA), conforme al Decreto 1072 de 2015, Art. 2.2.4.6.16 Literal 7 (Evaluación inicial del SG-SST) y Art. 2.2.4.6.12 Literal 4 (Documentación obligatoria del SG-SST). ⚠️ Atención: Al no autorizar el tratamiento de sus datos, el formulario quedará registrado únicamente con fines de control de asistencia. El trabajador puede revocar o conceder la autorización en cualquier momento comunicándolo por escrito al responsable de HSE o Recursos Humanos de ESQUIVENSA. El registro se guardará con estado "Sin autorización" y será diferenciado en los informes gerenciales.

    // EsquivenSA · Script · FM-H&S-004
// 1. Reemplace el ID de abajo con el ID de su Google Sheets
// 2. Guarde (Ctrl+S)

const SHEET_ID = 'PEGUE_AQUI_EL_ID_DE_SU_HOJA';
const HOJA_TRABAJADORES = 'Trabajadores';

function doPost(e) {
  try {
    var p = JSON.parse(e.postData.contents);
    if (p.action==='saveWorker')   return respuestaJSON(guardarTrabajador(p.data));
    if (p.action==='getAll')       return respuestaJSON(obtenerTodos());
    if (p.action==='deleteWorker') return respuestaJSON(eliminarTrabajador(p.data.cedula));
    if (p.action==='saveEMO')      return respuestaJSON(guardarExamen(p.data));
    return respuestaJSON({error:'Accion desconocida'});
  } catch(err) { return respuestaJSON({error:err.toString()}); }
}

function doGet(e) {
  var a = e&&e.parameter ? e.parameter.action||'' : '';
  if (a==='getAll') return respuestaJSON(obtenerTodos());
  if (a==='getEMO')  return respuestaJSON(obtenerExamenes());
  return respuestaJSON({status:'OK',mensaje:'EsquivenSA FM-H&S-004 OK'});
}

function guardarTrabajador(w) {
  var hoja=obtenerOCrearHoja(HOJA_TRABAJADORES), cab=obtenerCabecera(hoja);
  var ced=String(w.cedula||'').trim();
  if(!ced) return {error:'Cedula requerida'};
  var filas=hoja.getDataRange().getValues(), col=cab.indexOf('cedula'), fila=-1;
  for(var i=1;i<filas.length;i++) if(String(filas[i][col]).trim()===ced){fila=i+1;break;}
  var row=cab.map(function(c){
    var v=w[c]; if(Array.isArray(v))return v.join(', ');
    if(typeof v==='object'&&v)return JSON.stringify(v);
    return v!==undefined?String(v):'';
  });
  if(fila>0)hoja.getRange(fila,1,1,row.length).setValues([row]);
  else hoja.appendRow(row);
  return {ok:true,cedula:ced};
}

function obtenerTodos() {
  var hoja=obtenerOCrearHoja(HOJA_TRABAJADORES),d=hoja.getDataRange().getValues();
  if(d.length<2) return {workers:[]};
  var c=d[0];
  return {workers:d.slice(1).map(function(r){var o={};c.forEach(function(k,i){o[k]=r[i];});return o;})};
}

function eliminarTrabajador(cedula) {
  var hoja=obtenerOCrearHoja(HOJA_TRABAJADORES),d=hoja.getDataRange().getValues();
  var col=d[0].indexOf('cedula');
  for(var i=d.length-1;i>=1;i--) if(String(d[i][col]).trim()===String(cedula).trim()){hoja.deleteRow(i+1);return {ok:true};}
  return {error:'No encontrado'};
}

function obtenerOCrearHoja(nombre) {
  var libro=SpreadsheetApp.openById(SHEET_ID), hoja=libro.getSheetByName(nombre);
  if(!hoja){
    hoja=libro.insertSheet(nombre);
    hoja.appendRow(['cedula','nombre','nombre1','nombre2','apellido1','apellido2',
      'tipoDoc','fnac','edad','sexo','etnia','celular','email','contacto','parentesco',
      'telContacto','empresa','cargo','area','especialidad','lugarTrabajo','riesgo',
      'vinculacion','tipoContrato','horario','fechaIngreso','arl','eps','afp','salario',
      'experiencia','escolaridad','profesion','municipio','estrato','tipoVivienda',
      'estadoCivil','cabeFamily','tieneHijos','nHijos','rh','enfPre','alcohol','fuma',
      'actFisica','tieneMascotas','trabajaRemoto','bienestar','canal',
      'autorizaDatos','fechaEncuesta','respEncuestaNombre','estado','fechaRegistro']);
    hoja.setFrozenRows(1);
  }
  return hoja;
}

function obtenerCabecera(hoja){
  return hoja.getRange(1,1,1,hoja.getLastColumn()).getValues()[0].map(String);
}
function guardarExamen(examen) {
  var libro = SpreadsheetApp.openById(SHEET_ID);
  var hoja  = libro.getSheetByName('ExamenesMedicos');
  if (!hoja) {
    hoja = libro.insertSheet('ExamenesMedicos');
    hoja.appendRow(['cedula','nombre','cargo','empresa','tipo','fecha','ips','apto','obs']);
    hoja.setFrozenRows(1);
  }
  var datos = hoja.getDataRange().getValues();
  var colCed = 0, colFecha = 5, colTipo = 4;
  for (var i = 1; i < datos.length; i++) {
    if (String(datos[i][colCed])===String(examen.cedula) &&
        String(datos[i][colFecha])===String(examen.fecha) &&
        String(datos[i][colTipo])===String(examen.tipo)) {
      hoja.getRange(i+1,1,1,9).setValues([[
        examen.cedula,examen.nombre,examen.cargo,examen.empresa,
        examen.tipo,examen.fecha,examen.ips,examen.apto,examen.obs||''
      ]]);
      return {ok:true,accion:'actualizado'};
    }
  }
  hoja.appendRow([examen.cedula,examen.nombre,examen.cargo,examen.empresa,
    examen.tipo,examen.fecha,examen.ips,examen.apto,examen.obs||'']);
  return {ok:true,accion:'creado'};
}

function obtenerExamenes() {
  var libro = SpreadsheetApp.openById(SHEET_ID);
  var hoja  = libro.getSheetByName('ExamenesMedicos');
  if (!hoja) return {examenes:[]};
  var datos = hoja.getDataRange().getValues();
  if (datos.length < 2) return {examenes:[]};
  var cab = datos[0];
  return {examenes: datos.slice(1).map(function(r){
    var o={}; cab.forEach(function(c,i){o[c]=r[i];}); return o;
  })};
}

function respuestaJSON(obj){
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}
